def solution(enroll, referral, seller, amount):
    people = {}
    atomy = {}
    for i in range(len(enroll)):
        people[enroll[i]] = i
        atomy[enroll[i]] = referral[i]
    
    money = [0] * (len(enroll))
    for i in range(len(seller)):
        num = people[seller[i]] # ì…€ëŸ¬ì˜ ë²ˆí˜¸

        money[num] += cash   # ì¼ë‹¨ íŒ ëˆì„ ë°›ëŠ”ë‹¤
        while refer != '-' and cash > 0:

            money[num] -= give  # í—Œë‚©í•˜ëŠ” ëˆì„ ëº€ë‹¤
