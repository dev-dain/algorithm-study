def check():
  for i in range(N):
    line = i
    for j in range(H):
      if visited[j][line]:
        line += 1
      elif line > 0 and visited[j][line-1]:
        line -= 1
    if i != line:
      return False
  return True

def dfs(x, y, cnt):
  global res
  if check():
    res = min(res, cnt)
    return

    return

    k = y if i == x else 0
    for j in range(k, N-1):

  print(0)

  print(N)

res = 0
for com in combi:
  bit = 0
  for c in must:
    bit |= (1 << ord(c) - ord('a'))
  for c in com:
    bit |= (1 << ord(c) - ord('a'))
    
  cnt = 0
  for i in range(len(learn)):

      cnt += 1
  res = max(res, cnt)
